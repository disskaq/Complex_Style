61// Function to fill in address structure given an address and port
62static void fillAddr(const string &address, unsigned short port,
63 sockaddr_in &addr) {
64 memset(&addr, 0, sizeof(addr)); // Zero out address structure
65 addr.sin_family = AF_INET; // Internet address
66
67 hostent *host; // Resolve name
68 if ((host = gethostbyname(address.c_str())) == NULL) {
69 // strerror() will not work for gethostbyname() and hstrerror()
70 // is supposedly obsolete
71 throw SocketException("Failed to resolve name (gethostbyname())");
72 }
73 addr.sin_addr.s_addr = *((unsigned long *) host->h_addr_list[0]);
74
75 addr.sin_port = htons(port); // Assign port in network byte order
76}

80Socket::Socket(int type, int protocol) throw(SocketException) {
81 #ifdef WIN32
82 if (!initialized) {
83 WORD wVersionRequested;
84 WSADATA wsaData;
85
86 wVersionRequested = MAKEWORD(2, 0); // Request WinSock v2.0
87 if (WSAStartup(wVersionRequested, &wsaData) != 0) { // Load WinSock DLL
88 throw SocketException("Unable to load WinSock DLL");
89 }
90 initialized = true;
91 }
92 #endif
93
94 // Make a new socket
95 if ((sockDesc = socket(PF_INET, type, protocol)) < 0) {
96 throw SocketException("Socket creation failed (socket())", true);
97 }
98}

133void Socket::setLocalPort(unsigned short localPort) throw(SocketException) {
134 // Bind the socket to its port
135 sockaddr_in localAddr;
136 memset(&localAddr, 0, sizeof(localAddr));
137 localAddr.sin_family = AF_INET;
138 localAddr.sin_addr.s_addr = htonl(INADDR_ANY);
139 localAddr.sin_port = htons(localPort);
140
141 if (bind(sockDesc, (sockaddr *) &localAddr, sizeof(sockaddr_in)) < 0) {
142 throw SocketException("Set of local port failed (bind())", true);
143 }
144}

165unsigned short Socket::resolveService(const string &service,
166 const string &protocol) {
167 struct servent *serv; /* Structure containing service information */
168
169 if ((serv = getservbyname(service.c_str(), protocol.c_str())) == NULL)
170 return atoi(service.c_str()); /* Service is port number */
171 else
172 return ntohs(serv->s_port); /* Found port (network byte order) by name */
173}

298void UDPSocket::setBroadcast() {
299 // If this fails, we'll hear about it when we try to send. This will allow
300 // system that cannot broadcast to continue if they don't plan to broadcast
301 int broadcastPermission = 1;
302 setsockopt(sockDesc, SOL_SOCKET, SO_BROADCAST,
303 (raw_type *) &broadcastPermission, sizeof(broadcastPermission));
304}

306void UDPSocket::disconnect() throw(SocketException) {
307 sockaddr_in nullAddr;
308 memset(&nullAddr, 0, sizeof(nullAddr));
309 nullAddr.sin_family = AF_UNSPEC;
310
311 // Try to disconnect
312 if (::connect(sockDesc, (sockaddr *) &nullAddr, sizeof(nullAddr)) < 0) {
313 #ifdef WIN32
314 if (errno != WSAEAFNOSUPPORT) {
315 #else
316 if (errno != EAFNOSUPPORT) {
317 #endif
318 throw SocketException("Disconnect failed (connect())", true);
319 }
320 }

323void UDPSocket::sendTo(const void *buffer, int bufferLen,
324 const string &foreignAddress, unsigned short foreignPort)
325 throw(SocketException) {
326 sockaddr_in destAddr;
327 fillAddr(foreignAddress, foreignPort, destAddr);
328
329 // Write out the whole buffer as a single message.
330 if (sendto(sockDesc, (raw_type *) buffer, bufferLen, 0,
331 (sockaddr *) &destAddr, sizeof(destAddr)) != bufferLen) {
332 throw SocketException("Send failed (sendto())", true);
333 }

336int UDPSocket::recvFrom(void *buffer, int bufferLen, string &sourceAddress,
337 unsigned short &sourcePort) throw(SocketException) {
338 sockaddr_in clntAddr;
339 socklen_t addrLen = sizeof(clntAddr);
340 int rtn;
341 if ((rtn = recvfrom(sockDesc, (raw_type *) buffer, bufferLen, 0,
342 (sockaddr *) &clntAddr, (socklen_t *) &addrLen)) < 0) {
343 throw SocketException("Receive failed (recvfrom())", true);
344 }
345 sourceAddress = inet_ntoa(clntAddr.sin_addr);
346 sourcePort = ntohs(clntAddr.sin_port);
347
348 return rtn;
349}
